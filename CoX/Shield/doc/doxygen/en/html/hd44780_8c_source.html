63 void
64 HD44780Write(unsigned char ucRS, unsigned char ucInstruction)
65 {
66  //
67  // Check Arguments.
68  //
69  xASSERT((ucRS == HD44780_RS_COMMAND) || (ucRS == HD44780_RS_DATA));
70 
71  //
72  // RS:Command, RW:Write, E:Enable
73  //
74  xGPIOSPinWrite(HD44780_PIN_RS, ucRS);
75  xGPIOSPinWrite(HD44780_PIN_RW, HD44780_RW_WRITE);
76  xGPIOSPinWrite(HD44780_PIN_E, HD44780_E_ENABLE);
77 
78  //
79  // Output Data
80  //
81  xGPIOSPinWrite(HD44780_PIN_D7, (ucInstruction >> 7) & 0x01);
82  xGPIOSPinWrite(HD44780_PIN_D6, (ucInstruction >> 6) & 0x01);
83  xGPIOSPinWrite(HD44780_PIN_D5, (ucInstruction >> 5) & 0x01);
84  xGPIOSPinWrite(HD44780_PIN_D4, (ucInstruction >> 4) & 0x01);
85 
86  xGPIOSPinWrite(HD44780_PIN_E, HD44780_E_DISABLE);
87  xSysCtlDelay(10);
88  xGPIOSPinWrite(HD44780_PIN_E, HD44780_E_ENABLE);
89 
90  xGPIOSPinWrite(HD44780_PIN_D7, (ucInstruction >> 3) & 0x01);
91  xGPIOSPinWrite(HD44780_PIN_D6, (ucInstruction >> 2) & 0x01);
92  xGPIOSPinWrite(HD44780_PIN_D5, (ucInstruction >> 1) & 0x01);
93  xGPIOSPinWrite(HD44780_PIN_D4, ucInstruction & 0x01);
94 
95  xSysCtlDelay(10);
96 
97  //
98  // E:Disable
99  //
100  xGPIOSPinWrite(HD44780_PIN_E, HD44780_E_DISABLE);
101 }

116 unsigned char
117 HD44780Read(unsigned char ucRS)
118 {
119  unsigned char ucData = 0;
120 
121  //
122  // Check Arguments.
123  //
124  xASSERT((ucRS == HD44780_RS_COMMAND) || (ucRS == HD44780_RS_DATA));
125 
126  //
127  // Set D7 - D0 direction to GPIO Input
128  //
129  xGPIOSPinTypeGPIOInput(HD44780_PIN_D7);
130  xGPIOSPinTypeGPIOInput(HD44780_PIN_D6);
131  xGPIOSPinTypeGPIOInput(HD44780_PIN_D5);
132  xGPIOSPinTypeGPIOInput(HD44780_PIN_D4);
133 
134  //
135  // RS:Command, RW:Write, E:Enable
136  //
137  xGPIOSPinWrite(HD44780_PIN_RS, ucRS);
138  xGPIOSPinWrite(HD44780_PIN_RW, HD44780_RW_READ);
139  xGPIOSPinWrite(HD44780_PIN_E, HD44780_E_DISABLE);
140  xSysCtlDelay(10);
141  xGPIOSPinWrite(HD44780_PIN_E, HD44780_E_ENABLE);
142 
143  //
144  // Read the Data
145  //
146  ucData |= xGPIOSPinRead(HD44780_PIN_D7) << 7;
147  ucData |= xGPIOSPinRead(HD44780_PIN_D6) << 6;
148  ucData |= xGPIOSPinRead(HD44780_PIN_D5) << 5;
149  ucData |= xGPIOSPinRead(HD44780_PIN_D4) << 4;
150 
151  xGPIOSPinWrite(HD44780_PIN_E, HD44780_E_DISABLE);
152  xSysCtlDelay(10);
153  xGPIOSPinWrite(HD44780_PIN_E, HD44780_E_ENABLE);
154 
155  ucData |= xGPIOSPinRead(HD44780_PIN_D7) << 3;
156  ucData |= xGPIOSPinRead(HD44780_PIN_D6) << 2;
157  ucData |= xGPIOSPinRead(HD44780_PIN_D5) << 1;
158  ucData |= xGPIOSPinRead(HD44780_PIN_D4) << 0;
159 
160  //
161  // E:Disable
162  //
163  xGPIOSPinWrite(HD44780_PIN_E, HD44780_E_DISABLE);
164 
165  //
166  // At the End, set D7 - D0 direction to GPIO OutPut
167  //
168  xGPIOSPinTypeGPIOOutput(HD44780_PIN_D7);
169  xGPIOSPinTypeGPIOOutput(HD44780_PIN_D6);
170  xGPIOSPinTypeGPIOOutput(HD44780_PIN_D5);
171  xGPIOSPinTypeGPIOOutput(HD44780_PIN_D4);
172 
173  return ucData;
174 }

261 void
262 HD44780Init(void)
263 {
264  //
265  // Enable GPIO Port that used
266  //
267  xSysCtlPeripheralEnable(xGPIOSPinToPeripheralId(HD44780_PIN_D7));
268  xSysCtlPeripheralEnable(xGPIOSPinToPeripheralId(HD44780_PIN_D6));
269  xSysCtlPeripheralEnable(xGPIOSPinToPeripheralId(HD44780_PIN_D5));
270  xSysCtlPeripheralEnable(xGPIOSPinToPeripheralId(HD44780_PIN_D4));
271 
272  xSysCtlPeripheralEnable(xGPIOSPinToPeripheralId(HD44780_PIN_E));
273  xSysCtlPeripheralEnable(xGPIOSPinToPeripheralId(HD44780_PIN_RS));
274  xSysCtlPeripheralEnable(xGPIOSPinToPeripheralId(HD44780_PIN_RW));
275 
276  //
277  // Set Pins Type to GPIO Output
278  //
279  xGPIOSPinTypeGPIOOutput(HD44780_PIN_D7);
280  xGPIOSPinTypeGPIOOutput(HD44780_PIN_D6);
281  xGPIOSPinTypeGPIOOutput(HD44780_PIN_D5);
282  xGPIOSPinTypeGPIOOutput(HD44780_PIN_D4);
283 
284  xGPIOSPinTypeGPIOOutput(HD44780_PIN_E);
285  xGPIOSPinTypeGPIOOutput(HD44780_PIN_RS);
286  xGPIOSPinTypeGPIOOutput(HD44780_PIN_RW);
287 
288  //
289  // Output default value : E disable
290  //
291  xGPIOSPinWrite(HD44780_PIN_E, HD44780_E_DISABLE);
292 
293  //
294  // Set Entry Mode: Interface Data Length, Character Font, Display Line
295  //
296  while(HD44780Busy());
297  HD44780WriteCmd(HD44780_CMD_FUNCTION_SET(HD44780_INTERFACE_DATA_LEN |
298  HD44780_FUNCTION_SET_N_2 |
299  HD44780_CHARACTER_FONT));
300 
301  //
302  // Display on & Cursor Blink
303  //
304  while(HD44780Busy());
305  HD44780WriteCmd(HD44780_CMD_DISPLAY_CTRL(HD44780_DISPLAY_CTRL_D |
306  0 |
307  HD44780_DISPLAY_CTRL_B));
308 
309  //
310  // Clear LCD
311  //
312  while(HD44780Busy());
313  HD44780WriteCmd(HD44780_CMD_CLS);
314 
315  //
316  // Cursor Move Mode: Increment
317  //
318  while(HD44780Busy());
319  HD44780WriteCmd(HD44780_CMD_ENTRY_MODE_SET(HD44780_ENTRY_MODE_SET_ID_INC |
320  0));
321 
322 }
